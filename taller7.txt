# EJERCICIO 5

SELECT DISTINCT *
WHERE {
 ?person
            dbo:award ?prize ;
            dbo:birthName ?name .
 ?prize     dct:subject dbc:Nobel_Prize ;
            dbp:name  ?prizeName
}
ORDER BY ASC (?name)
LIMIT 100
