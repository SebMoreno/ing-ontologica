# EJERCICIO 1

PREFIX rob: <http://all-robotics.com#>

SELECT ?robot ?nombre 
FROM <Robots>
WHERE {
  ?robot
      a ?roboType ;
      rob:Identificacion ?id ;
      rob:Nombre ?nombre ;
      rob:Peso ?peso .
  ?roboType rdfs:subClassOf rob:Robots .
}
